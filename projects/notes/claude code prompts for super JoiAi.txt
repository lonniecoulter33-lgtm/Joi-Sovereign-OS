def call_llm_with_tools(prompt, provider="claude"):
    # Discover all available tools
    available_tools = []
    
    # Load tools from plugins
    for plugin in plugins:
        if hasattr(plugin, 'register_tools'):
            available_tools.extend(plugin.register_tools())
    
    # Include tools in LLM request
    if provider == "openai":
        response = openai.ChatCompletion.create(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}],
            functions=available_tools  # ← CRITICAL
        )
    elif provider == "claude":
        response = anthropic.messages.create(
            model="claude-sonnet-4",
            messages=[{"role": "user", "content": prompt}],
            tools=available_tools  # ← CRITICAL
        )
    
    # If LLM wants to call a tool
    if response.wants_to_call_function:
        tool_name = response.function_name
        tool_args = response.function_arguments
        
        # Execute the tool
        result = execute_tool(tool_name, **tool_args)
        
        # Send result back to LLM
        return call_llm_with_tools(result, provider)
    
    return response.content
